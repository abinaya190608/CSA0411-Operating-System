#include <stdlib.h>
int main() {
    int req[20], n, head, i, j, temp;
    int disk_size, direction;
    int total = 0;
    printf("Enter number of requests: ");
    scanf("%d", &n);
    printf("Enter the request queue: ");
    for (i = 0; i < n; i++)
        scanf("%d", &req[i]);
    printf("Enter initial head position: ");
    scanf("%d", &head);
    printf("Enter total disk size: ");
    scanf("%d", &disk_size);
    printf("Enter head direction (1 = towards higher cylinders, 0 = towards lower): ");
    scanf("%d", &direction);
    req[n] = head;
    n++;
    for (i = 0; i < n - 1; i++) {
        for (j = i + 1; j < n; j++) {
            if (req[i] > req[j]) {
                temp = req[i];
                req[i] = req[j];
                req[j] = temp;
            }
        }
    }
    int index = 0;
    for (i = 0; i < n; i++) {
        if (req[i] == head) {
            index = i;
            break;
        }
    }
    printf("\nSCAN Disk Scheduling\n");
    printf("Sequence of head movement:\n");
    if (direction == 1) {
        for (i = index; i < n; i++) {
            printf("%d → ", req[i]);
        }
        total += abs(req[n - 1] - head);
        printf("%d → ", disk_size - 1);
        total += abs((disk_size - 1) - req[n - 1]);
        for (i = index - 1; i >= 0; i--) {
            printf("%d → ", req[i]);
        }
        total += abs((disk_size - 1) - req[0]);
    }
    else {
        for (i = index; i >= 0; i--) {
            printf("%d → ", req[i]);
        }
        total += abs(head - req[0]);
        printf("0 → ");
        total += req[0];
        for (i = index + 1; i < n; i++) {
            printf("%d → ", req[i]);
        }
        total += abs(req[n - 1] - 0);
    }
    printf("\n\nTotal Head Movement = %d cylinders\n", total);
    return 0;
}
