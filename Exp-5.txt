#include <stdio.h>
int main() {
    int n, i;
    float avg_wt = 0, avg_tat = 0;
    printf("Enter number of processes: ");
    scanf("%d", &n);
    int bt[n], wt[n], tat[n];
    printf("Enter burst time for each process:
");
    for (i = 0; i < n; i++) {
        printf("Process %d: ", i + 1);
        scanf("%d", &bt[i]);
    }
    wt[0] = 0;  // Waiting time for first process is 0
    for (i = 1; i < n; i++) {
        wt[i] = wt[i - 1] + bt[i - 1]; // Waiting time = sum of burst times of all previous processes
    }
    for (i = 0; i < n; i++) {
        tat[i] = wt[i] + bt[i];  // Turnaround time = waiting time + burst time
        avg_wt += wt[i];
        avg_tat += tat[i];
    }
    printf("Process\tBurst Time\tWaiting Time\tTurnaround Time");
    for (i = 0; i < n; i++) {
        printf("P%d\t\t%d\t\t%d\t\t%d", i + 1, bt[i], wt[i], tat[i]);
    }
    printf("Average Waiting Time = %.2f", avg_wt / n);
    printf("Average Turnaround Time = %.2f", avg_tat / n);
    return 0;
}
