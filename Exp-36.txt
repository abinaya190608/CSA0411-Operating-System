#include <stdio.h>
struct Block {
    int blockNumber;
    int next;   
};
int main() {
    struct Block file[20];
    int n, i;
    printf("Enter number of blocks in the file: ");
    scanf("%d", &n);
    printf("Enter block numbers allocated to the file:\n");
    for (i = 0; i < n; i++) {
        printf("Block %d: ", i + 1);
        scanf("%d", &file[i].blockNumber);
        if (i == n - 1)
            file[i].next = -1;   
        else
            file[i].next = i + 1;
    }
    int start = 0;     
    int end = n - 1;  
    printf("\nLinked File Allocation:\n");
    printf("Start Block: %d\n", file[start].blockNumber);
    printf("End Block: %d\n\n", file[end].blockNumber);
    printf("BlockNumber → NextPointer\n");
    for (i = 0; i < n; i++) {
        printf("%d → ", file[i].blockNumber);
        if (file[i].next == -1)
            printf("NULL\n");
        else
            printf("%d\n", file[file[i].next].blockNumber);
    }
    printf("\nTraversing file blocks:\n");
    i = start;
    while (i != -1) {
        printf("Reading Block %d\n", file[i].blockNumber);
        i = file[i].next;  // move to next block
    }
    return 0;
}
