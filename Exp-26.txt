#include <stdio.h>
#include <stdlib.h>

int main() {
    FILE *fp;
    char filename[50] = "sample.txt";
    char ch;

    // 1. Create & Write into a file
    fp = fopen(filename, "w");
    if (fp == NULL) {
        printf("Error creating file!\n");
        exit(1);
    }
    fprintf(fp, "Hello! This is a sample file.\n");
    fprintf(fp, "File Management in C.\n");
    fclose(fp);
    printf("File created and data written successfully.\n\n");

    // 2. Read the file
    fp = fopen(filename, "r");
    if (fp == NULL) {
        printf("Error opening file for reading!\n");
        exit(1);
    }
    printf("Reading file content:\n");
    while ((ch = fgetc(fp)) != EOF)
        putchar(ch);
    fclose(fp);
    printf("\n\n");

    // 3. Append more data to the file
    fp = fopen(filename, "a");
    if (fp == NULL) {
        printf("Error opening file for appending!\n");
        exit(1);
    }
    fprintf(fp, "Appending more text to the file.\n");
    fclose(fp);
    printf("Data appended successfully.\n\n");

    // 4. Read again after append
    fp = fopen(filename, "r");
    printf("File content after appending:\n");
    while ((ch = fgetc(fp)) != EOF)
        putchar(ch);
    fclose(fp);
    printf("\n");

    // 5. Delete the file
    if (remove(filename) == 0)
        printf("File deleted successfully.\n");
    else
        printf("Error deleting file.\n");

    return 0;
}
