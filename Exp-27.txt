#include <stdio.h>
#include <stdlib.h>
#include <dirent.h>
void listFiles(const char *path) {
    struct dirent *dp;
    DIR *dir = opendir(path);
    if (dir == NULL) {
        return;
    }
    while ((dp = readdir(dir)) != NULL) {
        if (dp->d_name[0] == '.')
            continue;
        printf("%s\n", dp->d_name);
    }
    closedir(dir);
}
int main(int argc, char *argv[]) {
    if (argc == 1) {
        listFiles(".");
    }
    else {
        for (int i = 1; i < argc; i++) {
            printf("\n%s:\n", argv[i]);
            listFiles(argv[i]);
        }
    }
    return 0;
}
